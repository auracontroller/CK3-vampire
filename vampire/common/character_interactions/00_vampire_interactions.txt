create_ghoul_interaction = {
    category = interaction_category_diplomacy
    desc = create_ghoul_interaction_desc
    
    # The engine checks these conditions before even showing you the button
    is_shown = {
        scope:actor = { 
            has_trait = vampire 
        }
        scope:recipient = { 
            is_alive = yes 
            NOT = { has_trait = ghoul }
            NOT = { has_trait = vampire }
        }
    }
    
    # The engine checks this condition to determine if the button is clickable
    is_valid = {
        scope:recipient = {
            OR = {
                is_in_realm_of = scope:actor
                is_courtier_of = scope:actor
            }
        }
    }

    # Bypassing AI logic entirely
    auto_accept = yes

    # The synchronous execution of the variables
    on_accept = {
        scope:recipient = {
            add_trait = ghoul
            # Start the hunger cycle
            trigger_event = {
                id = vampire_ghoul.001
                days = 60
            }
        }
    }
}

ghoul_acquire_blood_interaction = {
    category = interaction_category_diplomacy
    desc = ghoul_acquire_blood_interaction_desc

    is_shown = {
        scope:actor = {
            has_trait = ghoul
            has_trait = need_blood
        }
        scope:recipient = {
            has_trait = vampire
        }
    }

    is_valid = {
        scope:recipient = {
            OR = {
                is_in_realm_of = scope:actor
                is_courtier_of = scope:actor
            }
        }
    }

    auto_accept = yes

    ai_will_do = {
        base = 0
        modifier = {
            add = 1000
            scope:actor = { has_trait = need_blood }
        }
    }

    on_accept = {
        scope:actor = {
            remove_trait = need_blood

            # Restart the hunger cycle
            trigger_event = {
                id = vampire_ghoul.001
                days = 60
            }

            # Strip all standard illness and injury traits explicitly (except scars)
            remove_trait = ill
            remove_trait = wounded_1
            remove_trait = wounded_2
            remove_trait = wounded_3
            remove_trait = maimed
            remove_trait = ill_pneumonia
            remove_trait = ill_typhus
            remove_trait = ill_consumption
            remove_trait = ill_cancer
            remove_trait = ill_gout
            remove_trait = ill_leprosy
            remove_trait = ill_lovers_pox
            remove_trait = ill_bubonic_plague
            remove_trait = ill_smallpox
            remove_trait = ill_great_pox
            remove_trait = ill_measles
            remove_trait = ill_dysentery
            remove_trait = ill_ergotism
            remove_trait = ill_dancing_plague
        }
    }
}

vampire_embrace_interaction = {
    category = interaction_category_diplomacy
    desc = vampire_embrace_interaction_desc

    is_shown = {
        scope:actor = {
            has_trait = vampire
        }
        scope:recipient = {
            is_alive = yes
            NOT = { has_trait = vampire }
        }
    }

    is_valid = {
        scope:recipient = {
            OR = {
                is_in_realm_of = scope:actor
                is_courtier_of = scope:actor
            }
        }
    }

    auto_accept = yes

    on_accept = {
        scope:recipient = {
            remove_trait = ghoul
            remove_trait = need_blood
            add_trait = vampire
            add_relation = {
                type = vampire_sire
                target = scope:actor
            }
        }
        scope:actor = {
            add_relation = {
                type = vampire_childe
                target = scope:recipient
            }
        }
    }
}
